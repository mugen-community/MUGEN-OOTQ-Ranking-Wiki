function applyFilters(lang) {
    const keyword = refs.search.value.trim().toLowerCase();
    const tierFilter = refs.tier.value;
    const trigSet = state.triggerFilter;
    const attSet = state.attackFilter;

    state.filtered = state.all.filter(item => {
      const tags = item.tagsByLang[lang] || item.tagsByLang[defaultLang] || [];
      const tagTexts = tags.map(t => tagLabel(t, lang).toLowerCase()).join(' ');
      const searchBlob = [item.slug, item.title, tagTexts].join(' ').toLowerCase();
      const tierOk = tierFilter === 'all' || item.tier === tierFilter;
      const keywordOk = keyword ? searchBlob.includes(keyword) : true;
      var trigOk = trigSet.size === 0;
      if (!trigOk) {
        trigOk = tags.some(function(t) { return t.category === 'trigger_technique' && trigSet.has(t.key); });
      }
      var attOk = attSet.size === 0;
      if (!attOk) {
        attOk = tags.some(function(t) { return (t.category === 'attack_technique' || t.category === 'effect_technique') && attSet.has(t.key); });
      }
      return tierOk && keywordOk && trigOk && attOk;
    }).sort((a, b) => {
      if (tierOrder[a.tier] !== tierOrder[b.tier]) return tierOrder[a.tier] - tierOrder[b.tier];
      return a.title.localeCompare(b.title, lang);
    });

    refs.resultCount.textContent = window.__('resultCount', lang).replace('{n}', state.filtered.length);
    renderCards(state.filtered, lang);
  }
